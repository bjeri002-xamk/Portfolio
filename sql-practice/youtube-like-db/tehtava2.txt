CREATE TABLE Kayttajat (
    id INTEGER PRIMARY KEY,
    nimi TEXT
);

CREATE TABLE Videot (
    id INTEGER PRIMARY KEY,
    nimi TEXT,
    kuvaus TEXT,
    kayttaja_id INTEGER REFERENCES Kayttajat,
    aika DATETIME,
    katselukerrat INTEGER
);

CREATE TABLE Peukut (
    id INTEGER PRIMARY KEY,
    video_id INTEGER REFERENCES Videot,
    kayttaja_id INTEGER REFERENCES Kayttajat,
    kommentti_id INTEGER REFERENCES Kommentit
    arvio BOOLEAN
);

CREATE TABLE Kommentit (
    id INTEGER PRIMARY KEY,
    video_id INTEGER REFERENCES Videot,
    kayttaja_id INTEGER REFERENCES Kayttajat,
    viesti TEXT,
    aika DATETIME
);

CREATE TABLE Kanavat (
    id INTEGER PRIMARY KEY,
    nimi TEXT,
    kayttaja_id INTEGER REFERENCES Kayttajat
);

CREATE TABLE KanavanVideot (
    id INTEGER PRIMARY KEY,
    kanava_id INTEGER REFERENCES Kanavat,
    video_id INTEGER REFERENCES Videot
)

CREATE TABLE Seuraamiset (
    id INTEGER PRIMARY KEY,
    kanava_id INTEGER REFERENCES Kanavat,
    kayttaja_id INTEGER REFERENCES Kayttajat
);

CREATE TABLE Viestit (
    id INTEGER PRIMARY KEY, 
    kayttaja1_id INTEGER REFERENCES Kayttajat, 
    kayttaja2_id INTEGER REFERENCES Kayttajat, 
    viesti TEXT, 
    aika DATETIME
);

CREATE TABLE Estot (
    id INTEGER PRIMARY KEY,
    estaja_id INTEGER REFERENCES Kayttajat,
    estetty_id INTEGER REFERENCES Kayttajat
);
